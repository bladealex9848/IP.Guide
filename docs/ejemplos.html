import requests

def get_my_ip():
    response = requests.get('https://ip.guide/')
    data = response.json()
    print(f"Tu IP: {data['ip']}")
    return data

def lookup_ip(ip: str):
    response = requests.get(f'https://ip.guide/{ip}')
    return response.json()

def lookup_asn(asn: int):
    response = requests.get(f'https://ip.guide/AS{asn}')
    return response.json()

# Uso
my_info = get_my_ip()

print(google['network']['autonomous_system']['name'])

üî∑ Go

Go
package main

import (
    "encoding/json"
    "fmt"
    "net/http"
    "io"
)

type IPInfo struct {
    IP       string `json:"ip"`
    Location struct {
        Country string `json:"country"`
    } `json:"location"`
}

func lookupIP(ip string) (*IPInfo, error) {
    resp, _ := http.Get("https://ip.guide/" + ip)
    defer resp.Body.Close()
    body, _ := io.ReadAll(resp.Body)
    
    var info IPInfo
    json.Unmarshal(body, &info)
    return &info, nil
}

func main() {
    info, _ := lookupIP("8.8.8.8")
    fmt.Printf("%s - %s\n", info.IP, info.Location.Country)
}
